optimization_model.py - Colab

import pulp as pl

# 定義最佳化問題：工地配置路徑最小化

model = pl.LpProblem("Site_Layout_Optimization", pl.LpMinimize)

distance = pl.LpVariable("Path_Distance", lowBound=0)

cost = pl.LpVariable("Material_Cost", lowBound=0)

# 目標函數：最小化總成本與距離

model += distance * 0.7 + cost * 0.3

print("優化模型計算中...")
